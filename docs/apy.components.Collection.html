Class: Collection

Collection

new Collection(name, service, endpoint, components)

A Collection of Resource(s)

Parameters:
Name Type Description
name string

Resource endpoint name
service apy.CompositeService

Service instance
endpoint string

REST API endpoint base
components Array

(optional) initial components
Example
var service = new apy.CompositeService(...);
var collection = new apy.components.Collection('myResourceName', service, 'myEndpoint');
function error(e) {
    console.log('Error', e);
}
function success() {
    console.log('Fetched Data', arguments);
}
function progress() {
    console.log('Progress', arguments);
}
collection.fetch(progress)
          .then(success)
          .catch(error);
